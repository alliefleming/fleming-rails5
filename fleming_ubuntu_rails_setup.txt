# Setting up Environment 
Ubuntu 16.04
https://www.digitalocean.com/community/tutorials/how-to-install-ruby-on-rails-with-rbenv-on-ubuntu-16-04

1. $ sudo apt-get update
2. $ sudo apt-get install autoconf bison build-essential libssl-dev libyaml-dev libreadline6-dev zlib1g-dev libncurses5-dev libffi-dev libgdbm3 libgdbm-dev
3. $ git clone https://github.com/rbenv/rbenv.git ~/.rbenv
4. $ echo 'export PATH="$HOME/.rbenv/bin:$PATH"' >> ~/.bashrc
5. $ echo 'eval "$(rbenv init -)"' >> ~/.bashrc
6. $ source ~/.bashrc
7. $ git clone https://github.com/rbenv/ruby-build.git ~/.rbenv/plugins/ruby-build
# list ruby versions
$ rbenv install -l

choosing to install ruby 2.4.2
8. $ rbenv install 2.4.2
9. $ rbenv global 2.4.2
10. $ echo "gem: --no-document" > ~/.gemrc
11. $ gem install bundler
# check gem location
$ gem env home
